#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>

#define BUFFERSIZE 4096
#define COPYMODE 0644

void oops(int pos, char *err);

int main(int argc, char *argv[])
{

    int old_fd, new_fd, n_char;
    char buf[BUFFERSIZE];

    // if(argc != 3) oops("missing file operand.\n", "argument");
    // 这样做会导致perror输出Sucess, 因为没有函数对errno赋值, 则errno保持0, 即没有错误

    if (argc != 3)
        fprintf(stderr, "missing file operand.\n");

    if ((old_fd = open(argv[1], O_RDONLY)) == -1)
        oops(__LINE__, "err open");

    if ((new_fd = creat(argv[2], COPYMODE)) == -1)
        oops(__LINE__, "err creat");

    while ((n_char = read(old_fd, buf, BUFFERSIZE)) > 0)
    {
        // if(write(new_fd, buf, BUFFERSIZE) != n_char) oops(__LINE__, "err write");
        // 这里写BUFFERSIZE的话会导致写入随机内容, 要与上面读到的字节数相同
        if (write(new_fd, buf, n_char) != n_char)
            oops(__LINE__, "err write");
    }

    // 因错误跳出循环的处理
    if (n_char == -1)
        oops(__LINE__, "err read");

    if ((close(old_fd) == -1) || (close(new_fd) == -1))
        oops(__LINE__, "err close");

    return 0;
}

void oops(int pos, char *err)
{
    fprintf(stderr, "./cp1.c:%d\t", pos);
    perror(err);
    exit(1);
}

注意:

  1. 在没有任何系统调用或者系统调用不出错的前提下, 使用perror会打印出"任意字符串":Sucess的内容;
  2. write(new_fd, buf, n_char) != n_char如果写成了write(new_fd, buf, BUFFERSIZE) != n_char会导致写入BUFFERSIZE-n_char的随机内容!
